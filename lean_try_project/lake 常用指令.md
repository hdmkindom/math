```bash
lake new your_project_name

# 或者手动创建一个新文件夹并在原地建立项目
mkdir your_folder_name && cd your_folder_name && lake init your_project_name
```

```bash
lake init project # 与git类似
```

```bash
lake install # 安装依赖

lake build  # 构建包

lake run # 运行项目,执行的 Main.lean 中代码

lake update # 更新依赖

lake clean # 清理构建文件

lake exe <模块名> # 运行主模块

lean --version # 查看版本


```
